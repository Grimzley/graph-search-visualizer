Pathfinding Visualizer

from js import document canvas = document.getElementById("grid") ctx = canvas.getContext("2d") ROWS, COLS = 20, 20 CELL_SIZE = 30 # draw a grid def draw_grid(): for i in range(ROWS): for j in range(COLS): ctx.strokeRect(j*CELL_SIZE, i*CELL_SIZE, CELL_SIZE, CELL_SIZE) def run_pathfinding(): draw_grid() ctx.fillStyle = "lightblue" ctx.fillRect(0, 0, CELL_SIZE, CELL_SIZE) ctx.fillRect(CELL_SIZE, CELL_SIZE, CELL_SIZE, CELL_SIZE) draw_grid()
